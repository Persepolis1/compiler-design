int factorial(int num){
   if (num == 0) then {
       return (1);
   }
   else {
       return (num * factorial(num-1));
   };
};

int number(){
    int x;
    x = 5;
    return (x);
};

int addNumbers(int a, int b, int c){
    int x;
    x = a + b + c;
    return (x);
};

program {
  int x;
  x = factorial(6) - 300 + number() + addNumbers(1,2,3);
  put(x);
};
